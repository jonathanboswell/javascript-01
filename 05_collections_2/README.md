### Object Interactions

Cookies + Milk

```javascript
var cookie = {
    type: 'chocolate chip',
    isDipped: false
}

var milk = {
    isDelicious: true,
    dip: function(cookie)
    {
        cookie.isDipped = true;
    }
}
```
